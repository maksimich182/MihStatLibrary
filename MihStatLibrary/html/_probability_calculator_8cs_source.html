118 public void Calculate(FreqHistogram freqHistogram)
119 {
120 if (freqHistogram.SzShift != freqHistogram.Dimension)
121 throw new ProbabilityCalculatorException("Количество единиц рассчитывается только при совпадении размерности и смещения!");
122
123 long onesCounter = 0;
124 int maxValueForNmOnesInByte = (Tools.ArNumberOne.Length <= freqHistogram.Histogram.Length)
125 ? Tools.ArNumberOne.Length : freqHistogram.Histogram.Length;
126
127 for(int i = 0; i < maxValueForNmOnesInByte; i++)
128 {
129 onesCounter += freqHistogram.Histogram[i] * Tools.ArNumberOne[i];
130 }
131
132 for(int i = maxValueForNmOnesInByte; i < freqHistogram.Histogram.Length; i++)
133 {
134 int value = i;
135 int onesInValue = 0;
136 while(value != 0)
137 {
138 onesInValue += (value & 0x1) == 1 ? 1 : 0;
139 value >>= 1;
140 }
141 onesCounter += freqHistogram.Histogram[i] * onesInValue;
142 }
143
144 _probabilityOne += (double)onesCounter / (freqHistogram.NmVectors * freqHistogram.Dimension);
145 _probabilityZero = 1 - _probabilityOne;
146 }
147
152 private void _calculate(BlockData blockData)
153 {
154 long count = 0;
155 foreach (var element in blockData.Data)
156 {
157 count += Tools.ArNumberOne[element];
158 }
159 _probabilityOne += (double)count / (blockData.SzBlockData * Tools.BITS_IN_BYTE);
160 _probabilityZero = 1 - _probabilityOne;
161 }
162 }
