147 public void Reduce(FreqHistogram destHistogram)
148 {
149 if (destHistogram.Dimension > this.Dimension)
150 throw new ReduceException("Размерность итоговой гистограммы больше размерности исходной!");
151
152 if (destHistogram._szShift != 1 || this._szShift != 1)
153 throw new ReduceException("Значения смещений исходной и итоговой гистограммы должны совпадать!");
154
155 destHistogram._histogram = new long[1 << destHistogram._dimension];
156
157 int maskOffset = this._dimension - destHistogram._dimension;
158 long reduceMask = destHistogram._mask << maskOffset;
159 for (int i = 0; i < this._histogram.Length; i++)
160 {
161 destHistogram._histogram[(i & reduceMask) >> maskOffset] += this._histogram[i];
162 }
163
164 destHistogram._nmVectors = this._nmVectors;
165
166 if (this._szRemain != 0)
167 {
168 destHistogram._remain = this._remain;
169 destHistogram._szRemain = this._szRemain;
170 destHistogram._calculateData(ref destHistogram._remain, ref destHistogram._szRemain);
171 }
