152 public void Reduce(MarkTable destTable)
153 {
154 if (destTable.Dimension > this.Dimension)
155 throw new ReduceException("Размерность итоговой гистограммы больше размерности исходной!");
156
157 if (destTable._szShift != 1 || this._szShift != 1)
158 throw new ReduceException("Значения смещений исходной и итоговой гистограммы должны совпадать!");
159
160 destTable._table = new long[1 << destTable._dimension];
161
162 int maskOffset = this._dimension - destTable._dimension;
163 long reduceMask = destTable._mask << maskOffset;
164 for (int i = 0; i < this._table.Length; i++)
165 {
166 destTable._table[(i & reduceMask) >> maskOffset] += this._table[i];
167 }
168
169 destTable._nmVectors = this._nmVectors;
170
171 if (this._szRemain != 0)
172 {
173 destTable._remain = this._remain;
174 destTable._szRemain = this._szRemain;
175 destTable._calculateData(ref destTable._remain, ref destTable._szRemain);
176 }
