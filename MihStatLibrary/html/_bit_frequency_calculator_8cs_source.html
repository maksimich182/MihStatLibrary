118 public void Calculate(MarkTable markTable)
119 {
120 if (markTable.SzShift != markTable.Dimension)
121 throw new BitFrequencyCalculatorException("Количество единиц рассчитывается только при совпадении размерности и смещения!");
122
123 long onesCounter = 0;
124 int maxValueForNmOnesInByte = (Tools.ArNumberOne.Length <= markTable.Table.Length)
125 ? Tools.ArNumberOne.Length : markTable.Table.Length;
126
127 for(int i = 0; i < maxValueForNmOnesInByte; i++)
128 {
129 onesCounter += markTable.Table[i] * Tools.ArNumberOne[i];
130 }
131
132 for(int i = maxValueForNmOnesInByte; i < markTable.Table.Length; i++)
133 {
134 int value = i;
135 int onesInValue = 0;
136 while(value != 0)
137 {
138 onesInValue += (value & 0x1) == 1 ? 1 : 0;
139 value >>= 1;
140 }
141 onesCounter += markTable.Table[i] * onesInValue;
142 }
143
144 _frequencyOne += (double)onesCounter / (markTable.NmVectors * markTable.Dimension);
145 _frequencyZero = 1 - _frequencyOne;
146 }

152 public void Calculate(BlockData blockData)
153 {
154 long count = 0;
155 foreach (var element in blockData.Data)
156 {
157 count += Tools.ArNumberOne[element];
158 }
159 _frequencyOne = (double)count / (blockData.SzBlockData * Tools.BITS_IN_BYTE);
160 _frequencyZero = 1 - _frequencyOne;
161 }
